编写一个程序，让用户输入其年龄，然后显示该年龄包含多少个月

#include <iostream>
int ageToMonth(int);


int main()
{
    using namespace std;
    int age;
    cout << "Enter your age: ";
    cin >> age;
    int month = ageToMonth(age);
    cout << "Your age in months is " << month << endl;
    return 0;
}

int ageToMonth(int a)
{
    return a * 12;
}

编写一个程序，其中的main()调用一个用户定义的函数，以摄氏温度值为参数，返回相应的华氏温度值

#include <iostream>
double Tem(double);

int main()
{
    using namespace std;
    double S;
    cout << "Please enter a Celsius value: ";
    cin >> S;
    cout << endl;
    double H = Tem(S);
    cout << S << " degrees Celsius is " << H << " degrees Fahrenheit."; 
    return 0;
}

double Tem (double x)
{
    return x * 1.8 + 32.0;
}

编写一个程序，要求用户输入小时数和分钟数。在main()函数中，将这两个值传递给一个void函数

#include <iostream>
void Time(int,int);

int main()
{
    using namespace std;
    int Hour;
    int Min;
    cout << "Enter the number of hours: ";
    cin >> Hour;
    cout << endl;
    cout << "Enter the number of minutes: ";
    cin >> Min;
    cout << endl;
    Time(Hour,Min);
    return 0;
}

void Time(int H,int M)
{
    using namespace std;
    cout << "Time: " << H << ":" << M;
}
